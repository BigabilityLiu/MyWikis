第一种方式，直接取得XCode目录中的.plist文件,这个文件应该是自己手动创建或添加的<br>
```Swift
let pathBundle = NSBundle.mainBundle().pathForResource("UserList", ofType: "plist")
                    print(pathBundle!)
```
<br>输出结果为<br>
```Swift
/Users/LiusMac/Library/Developer/CoreSimulator/Devices/51C2CCD9-FE37-4D64-A272-389F3CF17AFA/data/Containers/Bundle/Application/37C9C0D6-E0E5-4FDE-BBC8-14AF375DCE4F/TayoOffice.app/UserList.plist
```
<br>第二种方式，获得的就是所谓的沙盒目录<br>
```Swift
let documentsDirectorys = NSSearchPathForDirectoriesInDomains(NSSearchPathDirectory.DocumentDirectory, NSSearchPathDomainMask.UserDomainMask, true) as NSArray
let documentsDirectory = documentsDirectorys.objectAtIndex(0) as! NSString
let pathDocuments = documentsDirectory.stringByAppendingPathComponent("UserList.plist") as String
print(pathDocuments)
```
<br>输出结果<br>
```Swift
/Users/LiusMac/Library/Developer/CoreSimulator/Devices/51C2CCD9-FE37-4D64-A272-389F3CF17AFA/data/Containers/Data/Application/A80045CB-75FF-4A06-B2D9-CB9FEB9B7478/Documents/UserList.plist
```
<br>保存图片到本地<br>
```Swift
func saveImage(image: UIImage, filename: String) {
    let path = NSHomeDirectory().stringByAppendingString("/Documents/\(filename)")
    let data = UIImagePNGRepresentation(image)
    data.writeToFile(path, atomically: true)
}
```
<br>从本地取出图片<br>
```Swift
func getImage(filename: String) -> UIImage? {
    var error: NSError?
    let path = NSHomeDirectory().stringByAppendingString("/Documents/\(filename)")
    let data = NSData(contentsOfFile: path, options: .UncachedRead, error: &error)
    if let unwrappedError = error {
        return nil
    } else {
        return UIImage(data: data!)
    }
}
```
